    })

     "name": "nyc",

             {"@key":"database","$":"nyc"},
             {"@key":"user","$":"bob"},
             {"@key":"passwd","$":"postgres"},
             {"@key":"dbtype","$":"postgis"}
         ]
      }
    }).then(res => {
     console.log(res)
    })

     "name": "nyc",

    }).then(res => {
     console.log(res)
    })
